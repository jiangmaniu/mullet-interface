
config.json 用例：
```json
{
  "name": "Mullet", // 平台名稱
  "platform": "sux", // 平台代碼
  "websiteUrl": "www.mullet.io", // 網站域名
  "CLIENT_ID": "trade-pc-client", // 客戶端ID
  "CLIENT_SECRET": "trade-pc-client-secret", // 客戶端密碼
  "REGISTER_APP_CODE": "123456", // 註冊應用碼
  "tradingViewUrl": "https://tradingview.mullet.io", // 交易視圖地址
  "ws": "wss://websocket.mullet.io/websocketServer", // websocket地址
  "imgDomain": "https://file.mullet.io/trade/", // 圖片域名
  "REGISTER_MODULE": true, // 開啟註冊模組
  "PAYMENT_MODULE": true, // 開啟支付模組
  "SKIP_KYC_STEP_ONE": true, // 允许跳過KYC第一步
  "ID_CARD_ONLY": true // 只支持身份证认证
}
```
